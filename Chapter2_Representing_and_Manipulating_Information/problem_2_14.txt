x = 0x66  = (bin)0110 0110
y = 0x39  = (bin)0011 1001
~x        = (bin)1001 1001
~y        = (bin)1100 0110

x & y     = (bin)0010 0000 = 0x20
x | y     = (bin)0111 1111 = 0x7F
~x | ~ y  = (bin)1101 1111 = 0xDF
x & !y    = (bin)0000 0000 = 0x00

x && y    = 0x01
x || y    = 0x01
!x || ! y = 0x00
x && ~y   = 0x00
